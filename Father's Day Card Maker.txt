<script>
    const canvas = document.getElementById('dadCanvas');
    const ctx = canvas.getContext('2d');
    
    let uploadedImage = null;
    let selectedFrameIdx = 0;

    // Premium Masculine Design Configurations
    const frames = [
        { name: "Classic Navy", bg: "#F0F4F8", border: "#1E3A8A", accent: "#60A5AF" },
        { name: "Executive Gold", bg: "#FCFBF7", border: "#B45309", accent: "#D4AF37" },
        { name: "Forest Minimal", bg: "#F4F7F5", border: "#064E3B", accent: "#A7F3D0" },
        { name: "Vintage Charcoal", bg: "#F1F5F9", border: "#334155", accent: "#94A3B8" }
    ];

    // Image Upload Listener Pipeline
    document.getElementById('imageUpload').addEventListener('change', function(e) {
        const reader = new FileReader();
        reader.onload = function(event) {
            const img = new Image();
            img.onload = function() {
                uploadedImage = img;
                drawCard();
            };
            img.src = event.target.result;
        };
        if(e.target.files[0]) {
            reader.readAsDataURL(e.target.files[0]);
        }
    });

    function changeFrame(index) {
        document.querySelector('.dad-item.active').classList.remove('active');
        document.getElementsByClassName('dad-item')[index].classList.add('active');
        selectedFrameIdx = index;
        drawCard();
    }

    // Canvas Processing Loop
    function drawCard() {
        const frame = frames[selectedFrameIdx];
        const greetingText = document.getElementById('userGreeting').value || "";

        // 1. Core Card Canvas Background
        ctx.fillStyle = frame.bg;
        ctx.fillRect(0, 0, 600, 600);

        // 2. Compute Smart Proportional Cropping Vector
        if (uploadedImage) {
            ctx.save();
            
            const imgX = 120;
            const imgY = 100;
            const imgSize = 360;

            // Clean Rounded Rectangle Clipping Node
            ctx.beginPath();
            ctx.roundRect(imgX, imgY, imgSize, imgSize, 12);
            ctx.clip();

            // Mathematically maintain aspect ratio while centering
            const minSide = Math.min(uploadedImage.width, uploadedImage.height);
            const sx = (uploadedImage.width - minSide) / 2;
            const sy = (uploadedImage.height - minSide) / 2;

            ctx.drawImage(uploadedImage, sx, sy, minSide, minSide, imgX, imgY, imgSize, imgSize);
            ctx.restore();

            // Fine Contrast Border Profile
            ctx.strokeStyle = frame.border;
            ctx.lineWidth = 4;
            ctx.beginPath();
            ctx.roundRect(imgX, imgY, imgSize, imgSize, 12);
            ctx.stroke();
        } else {
            // Interactive UI Empty State Block
            ctx.fillStyle = frame.accent;
            ctx.globalAlpha = 0.3;
            ctx.beginPath();
            ctx.roundRect(120, 100, 360, 360, 12);
            ctx.fill();
            ctx.globalAlpha = 1.0; // Reset alpha

            ctx.fillStyle = frame.border;
            ctx.font = 'italic 16px "Segoe UI", sans-serif';
            ctx.textAlign = 'center';
            ctx.fillText("[ Upload Dad's Photo to Preview ]", 300, 290);
        }

        // 3. Structural Border Accents
        ctx.strokeStyle = frame.border;
        ctx.lineWidth = 8;
        ctx.strokeRect(25, 25, 550, 550);
        
        ctx.strokeStyle = frame.accent;
        ctx.lineWidth = 2;
        ctx.strokeRect(36, 36, 528, 528);

        // Corner Accents (Corner Crosshair Lines for a Structured Look)
        ctx.strokeStyle = frame.border;
        ctx.lineWidth = 3;
        const positions = [25, 575];
        
        // 4. Custom Message Layout Processing
        ctx.textAlign = 'center';
        ctx.fillStyle = frame.border;
        ctx.font = 'bold 23px "Segoe UI", Arial, sans-serif';

        const words = greetingText.split(' ');
        let line = '';
        let startY = 510;
        const lineHeight = 32;

        for (let n = 0; n < words.length; n++) {
            let testLine = line + words[n] + ' ';
            let metrics = ctx.measureText(testLine);
            if (metrics.width > 480 && n > 0) {
                ctx.fillText(line, 300, startY);
                line = words[n] + ' ';
                startY += lineHeight;
            } else {
                line = testLine;
            }
        }
        ctx.fillText(line, 300, startY);
    }

    // Dynamic text entry tracking
    document.getElementById('userGreeting').addEventListener('input', drawCard);

    // Save File Generation Engine
    document.getElementById('downloadDadBtn').addEventListener('click', function() {
        const link = document.createElement('a');
        link.download = `Fathers_Day_Card.png`;
        link.href = canvas.toDataURL('image/png');
        link.click();
    });

    // Mount initial triggers
    window.onload = () => { drawCard(); };
    setTimeout(drawCard, 300);
</script>
